Step 1

We are going to create a very simple Promise object that contains two functions

then: receives a function that will be executed when the promise is resolved

resolve: receives the value of the promises and call the callback received in the then function


          var Q = function() {
            var _then;
            this.then = function(callback) {
                _then = callback;
            }
            this.resolve = function(value) {
                _then(value);
            }
          }

Great! We have our fisrt promise object :)

Test it:

To test this we should create an Async action that returns a promise:

          var AsyncAction = function(action) {
            return function(){
                var args = Array.prototype.slice.call(arguments);
                var promise = new Q();
                setTimeout(function() {
                    var value = action.apply(this, args);
                    promise.resolve(value);
                },1000);
                return promise;
            };
          }

          var multiply = AsyncAction(function(arg, arg1) {
            console.log('Args:', arg);
            return arg*arg1
          });

          var p = multiply(2,4);
          p.then(function(result) {
            console.log(result);
          });
