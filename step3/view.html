Step 3

A Promise has a status: the values can be "pending", "resolved", "rejected"


          var Q = function() {
            var self = this;
            this.status = "pending";
            var _then, _reject;
            this.then = function(callback, reject) {
                _then = callback;
                _reject = reject;
            };
            this.resolve = function(value) {
                self.status= "resolved";
                _then(value);
            };
            this.reject = function(error) {
                self.status = "rejected";
                _reject(error);
            };
          }

Great! now we the status of a promise :)

Test it:

To test this we should create an Async action that returns a promise:

          var AsyncAction = function(action) {
            return function(){
                var args = Array.prototype.slice.call(arguments);
                var promise = new Q();
                setTimeout(function() {
                    try {
                      var value = action.apply(this, args);
                      promise.resolve(value);
                    }
                    catch (e) {
                      promise.reject(e);
                    }
                },1000);
                return promise;
            };
          }

          var multiply = AsyncAction(function(arg, arg1) {
            return arg * arg1
          });

          var p = multiply(2,4);
          console.log('Status:' + p.status);
          p.then(function(result) {
            console.log(result);
            console.log('Status:' + p.status);
          },function(error) {
            console.log(error);
            console.log('Status:' + p.status);
          });
          //Returns: Status: pending
          //         Status: resolved
